# No Parameters

func()

==>

Recipe(
  Call(QualifiableIdentifier(Identifier), ArgList)
)

# One Variable

f( x )

==>

Recipe(
  Call(QualifiableIdentifier(Identifier), ArgList(
    Variable(Identifier)
  ))
)

# Two Variables

f(x,y)

==>

Recipe(
  Call(QualifiableIdentifier(Identifier), ArgList(
    Variable(Identifier),
    Variable(Identifier)
  ))
)

# Three Variables

f(x, y, z)

==>

Recipe(
  Call(
    QualifiableIdentifier(Identifier),
    ArgList(
      Variable(Identifier),
      Variable(Identifier),
      Variable(Identifier)
    )
  )
)

# Nested Call

f(g())

==>

Recipe(
  Call(QualifiableIdentifier(Identifier), ArgList(
    Call(QualifiableIdentifier(Identifier), ArgList)
  ))
)

# Nested Nested Call

f( g( h() ) )

==>

Recipe(
  Call(QualifiableIdentifier(Identifier), ArgList(
    Call(QualifiableIdentifier(Identifier), ArgList(
      Call(QualifiableIdentifier(Identifier), ArgList)
    ))
  ))
)

# Variable and Call

f ( x , g ( ) )

==>

Recipe(
  Call(QualifiableIdentifier(Identifier), ArgList(
    Variable(Identifier),
    Call(QualifiableIdentifier(Identifier), ArgList)
  ))
)

# Call and Variable

f (
  g ( ) ,
  y
)

==>

Recipe(
  Call(QualifiableIdentifier(Identifier), ArgList(
    Call(QualifiableIdentifier(Identifier), ArgList),
    Variable(Identifier)
  ))
)

# Nested Nested Call with Variable Parameters

f(
  g(
    h(x,y),
    z
  )
)

==>

Recipe(
  Call(QualifiableIdentifier(Identifier), ArgList(
    Call(QualifiableIdentifier(Identifier), ArgList(
      Call(
        QualifiableIdentifier(Identifier),
        ArgList(Variable(Identifier), Variable(Identifier))
      ),
      Variable(Identifier)
    ))
  ))
)
