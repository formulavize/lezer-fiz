# Single Variable Alias

x = y

==>

Recipe(
  Alias(
    LhsVariable(Variable(Identifier)),
    Eq,
    Variable(Identifier)
  )
)

# Single Variable Assignment

foo = f()

==>

Recipe(
  Assignment(
    LhsVariable(Variable(Identifier)),
    Eq,
    Call(QualifiableIdentifier(Identifier), ArgList)
  )
)

# Unpacked Variable List Assignment

a, b, c = g()

==>

Recipe(
  Assignment(
    LhsVariable(Variable(Identifier)),
    LhsVariable(Variable(Identifier)),
    LhsVariable(Variable(Identifier)),
    Eq,
    Call(QualifiableIdentifier(Identifier), ArgList)
  )
)
